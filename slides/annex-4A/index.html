1. Contexto: Tipos avanzados

15 minutos

El problema: Tipos inflexibles

interface Task {
  id: string;
  title: string;
  status: 'pending' | 'inProgress' | 'completed';
}

// ¿Cómo creo un tipo para tarea PARCIALMENTE editable?
// ¿Cómo creo un tipo sin el campo id?
// ¿Cómo creo un tipo solo con los campos de búsqueda?

Solución: Utility Types

type TaskPartial = Partial<Task>;     // Todos los campos opcionales
type TaskSinId = Omit<Task, 'id'>;    // Sin el campo id
type TaskPreview = Pick<Task, 'id' | 'title'>; // Solo algunos campos
type TaskMap = Record<string, Task>;  // Objeto con claves string

2. Utility Types

25 minutos

Partial, Required, Omit, Pick

UtilityUso
Partial<T>Todos los campos opcionales
Required<T>Todos los campos obligatorios
Omit<T, K>Excluir campos
Pick<T, K>Seleccionar campos
Record<K,V>Índice/Map

3. Generic Constraints

20 minutos

extends para limitar genéricos

function getProp<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const task = { title: 'Learn TS', priority: 'high' };
getProp(task, 'title');    // ✅ string
getProp(task, 'priority'); // ✅ string
getProp(task, 'status');   // ❌ Error: no existe

5. Mapped Types

15 minutos

Transformar tipos automáticamente

type Nullable<T> = { [K in keyof T]: T[K] | null };

6. Type Inference

10 minutos

Inferencia en funciones

// TS infiere el tipo de retorno
function getTask() {
  return { id: '1', title: 'Learn TS' };
}
// Infiere: { id: string; title: string }

// as const - preservar literales
const colors = ['red', 'blue'] as const;
// readonly ['red', 'blue']

7. Satisfies Operator

10 minutos

satisfies vs as

// as - pierde literal
const c1 = 'debug' as string; // c1: string

// satisfies - mantiene literal + valida
const c2 = 'debug' satisfies 'debug' | 'info' | 'error';
// c2: 'debug' (literal)

8. Migración JSX → TSX

15 minutos

De JSX a TSX

// TSX - con tipos explícitos
interface BookCardProps {
  titulo: string;
  autor: string;
  año?: number;
}

function BookCard({ titulo, autor, año }: BookCardProps) {
  return (
    <div>
      <h3>{titulo}</h3>
      <p>{autor} {año && `- ${año}`}</p>
    </div>
  );
}

13. Componentes Genéricos

15 minutos

Componentes con tipo genérico

interface ListProps<T> {
  items: T[];
  renderItem: (item: T) => ReactNode;
}

function List<T>({ items, renderItem }: ListProps<T>) {
  return <ul>{items.map(renderItem)}</ul>;
}
